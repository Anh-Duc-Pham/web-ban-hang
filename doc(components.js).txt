components.js
  --params:
    -components: sử dụng querySelectorAll để lấy các component
    -wrapElement: sử dụng selector để lấy element bọc bên ngoài (mặc định là body)
    -có thể dùng phương thức khởi tạo
      vd: components.wrapElement = ".abc";
  --create: render các component
  --attributes component:
    -dbName: tên cơ sở dữ liệu (Bắt buộc)
    -parentElement: định dạng của thẻ cha (mặc định là thẻ div)
    -class: class của thẻ cha
  --template:
    -bản thiết kế phải viết bên trong thẻ component
    -dữ liệu trả về luôn là item
    vd:
      <abcComponent dbName="users" parentElement="ul" class="abc">
        <li>${item.name}</li>
      </abcComponent>
